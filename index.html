What is canvas?

Canvas is an HTML element which can be used to draw graphics via scripting (usually JavaScript).

Basic usage of canvas

By default, the canvas size is 300x150px

Getting Started

						
							const canvas = document.getElementById("myCanvas");
							const ctx = canvas.getContext("2d");

Drawing Basic Shapes

						
							// Drawing a rectangle
							ctx.fillRect(50, 50, 100, 75);

							// Drawing a circle
							ctx.beginPath();
							ctx.arc(200, 150, 50, 0, Math.PI * 2);
							ctx.fill();

							// Drawing a line
							ctx.beginPath();
							ctx.moveTo(50, 250);
							ctx.lineTo(250, 250);
							ctx.stroke();

							// Drawing a triangle
							ctx.beginPath();
							ctx.moveTo(350, 50);
							ctx.lineTo(300, 125);
							ctx.lineTo(400, 125);
							ctx.closePath();
							ctx.fill();

							// Drawing an ellipse
							ctx.beginPath();
							ctx.ellipse(400, 250, 80, 40, 0, 0, Math.PI * 2);
							ctx.fill();

Colors and Styles

						
							// Setting fill color
							ctx.fillStyle = "blue";
							ctx.fillRect(50, 50, 100, 75);

							// Setting stroke color
							ctx.strokeStyle = "red";
							ctx.strokeRect(200, 50, 100, 75);

Gradients and Patterns

						
							// Creating a linear gradient
							const gradient = ctx.createLinearGradient(50, 200, 200, 250);
							gradient.addColorStop(0, "blue");
							gradient.addColorStop(1, "white");

							ctx.fillStyle = gradient;
							ctx.fillRect(50, 200, 150, 50);

Drawing Text

						
							ctx.font = "20px Arial";
							ctx.fillStyle = "red";
							ctx.fillText("Hello, Canvas!", 50, 350);

Animations with Canvas

						
							const ballRadius = 20;
							let x = ballRadius + 5;
							let dx = 2;

							function drawBall() {
							  ctx.beginPath();
							  ctx.arc(x, canvas.height / 2, ballRadius, 0, Math.PI * 2);
							  ctx.fillStyle = "blue";
							  ctx.fill();
							  ctx.closePath();
							}

							function animateBall() {
							  ctx.clearRect(0, 0, canvas.width, canvas.height);
							  drawBall();
							  if (x+dx > canvas.width - ballRadius || x+dx < ballRadius){
								dx = -dx;
							  }
							  x += dx;
							  requestAnimationFrame(animateBall);
							}
							animateBall();

Interactivity with Canvas

						
							canvas.addEventListener("click", (event) => {
							  const x = event.clientX - canvas.offsetLeft;
							  const y = event.clientY - canvas.offsetTop;
							  ctx.fillRect(x - 25, y - 25, 50, 50);
							});
